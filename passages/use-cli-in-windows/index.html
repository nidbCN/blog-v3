function Get-Webp {
    param(
        [string] $FileName,
        [int] $Width = -1,
        [int] $Height = -1
    )

    $file = Get-Item $FileName;

    if (($Width -ne -1) -and ( $Height -ne -1) ) {
        ffmpeg -i $file.FullName $file.Name.Replace($file.Extension, ".webp");
    }
    else {
        ffmpeg -i $file.FullName -vf scale=${Width}:${Height} $file.Name.Replace($file.Extension, ".webp");
    }
}
