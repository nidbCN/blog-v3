struct ListNode *reverseList(struct ListNode *head)
{
    struct ListNode *lastNode = head;
    struct ListNode *temp = head;

    if (head != NULL && head->next != NULL)
    {
        for (int isFirst = 1; head->next != NULL; temp = head)
        {
            head = head->next;
            if (isFirst == 1)
            {
                temp->next = NULL;
                isFirst = 0;
            }
            else
            {
                temp->next = lastNode;
            }
            lastNode = temp;
        }
        head->next = lastNode;
    }
    return head;
}
